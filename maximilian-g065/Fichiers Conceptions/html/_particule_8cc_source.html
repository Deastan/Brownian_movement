132 void Particule::collision(Particule& p, GenerateurAleatoire& tirage)
133 {
134  Vecteur vg (masse/(masse + p.masse) * vitesse + p.masse/(masse + p.masse) * p.vitesse);
135 
136  double L (sqrt((vitesse-vg) * (vitesse-vg))); // racine carrée du produit scalaire d'un vecteur avec lui-même = norme de ce vecteur
137 
138  double z(tirage.uniforme_reel(-L, L));
139  double phi(tirage.uniforme_reel(0, 2 * M_PI));
140 
141  double r (sqrt(L * L - z * z));
142  Vecteur v0 (r * cos(phi), r * sin(phi), z);
143 
144  vitesse = vg + v0;
145  p.vitesse = vg - (masse/p.masse) * v0;
146 }
